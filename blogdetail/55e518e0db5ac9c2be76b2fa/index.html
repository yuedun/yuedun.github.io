使用nodemailer发送邮件

月盾
2015-09-01 03:17
Nodejs
目录

var nodemailer = require(&#39;nodemailer&#39;);

var transporter = nodemailer.createTransport({

service: &#39;gmail&#39;,

auth: {

user: &#39;youname@gmail.com&#39;,

pass: &#39;password&#39;

}

});

transporter.sendMail({

from: &#39;youname@gmail.com&#39;,

to: &#39;username@qq.com&#39;,

subject: &#39;hello&#39;,

text: &#39;hello world!&#39;

});
