使用nodemailer发送邮件

月盾
2015-09-01 03:17
后端 
var nodemailer = require("nodemailer");

var transporter = nodemailer.createTransport({

	service: "gmail",

	auth: {

		user: "youname@gmail.com",

		pass: "password"

	}

});

transporter.sendMail({

	from: "youname@gmail.com",

	to: "username@qq.com",

	subject: "hello",

	text: "hello world!"

});
